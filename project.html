ebola <- 
  read.csv("https://ds-ec2.scraperwiki.com/g7nnqgn/ckm9nsfssakeuor/cgi-bin/csv/ebola_data_db_format.csv")

Cumulative number of confirmed Ebola cases

ebola1 <-
  ebola %>%
  filter(Indicator == "Cumulative number of confirmed Ebola cases", value > 100, Date %in% c("2015-12-29", "2014-12-31", "2016-03-23"))

ebola1 %>%
  ggplot(aes(Date, value)) + geom_col()+ facet_wrap(~Country)

Read in Ebola data table

Gender <-
  read.csv("CASES.csv",header = TRUE, sep=",")

Calculate the total and mean affected male by Ebola

EbolaMale <-
  Gender %>%
  filter(Gender == "M")%>%
  summarise(Count=sum(Total), Average = mean(Total))
